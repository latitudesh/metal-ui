This component is an inline editable text input. It inherits typographic properties from its parent html element.

# Default

export const EditableWithState = () => {
    const [value, setValue] = useState('Click to Edit')
    return (
    <h1 className='text-2xl'>
        <Editable value={value} onSave={x => {
            setValue(x)
            return true
        }}/>
        </h1>
    )
}
