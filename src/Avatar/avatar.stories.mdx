import { Meta, Story, Preview, Props } from "@storybook/addon-docs/blocks";
import Avatar from "./Avatar";
import Text from "../Typography/Text";

export const names = [
  "Cheryl Carter",
  "Heather Morales",
  "Sean Jackson",
  "Catherine Anderson",
  "Jack Phillips",
  "Julia Williamson",
  "Jonathan Martin",
  "Kevin Niparko",
];

export const anonymousIds = [1591, 13184, 1055, 4199, 4824, 11394, 1965, 13023];

export const colors = [
  "neutral",
  "blue",
  "red",
  "orange",
  "yellow",
  "green",
  "teal",
  "purple",
];

# API

<Props of={Avatar} />

<Meta title="Avatar" component={Avatar} />

# Basic

Avatar with the <code>name</code> and <code>size</code> properties.

<Preview>
  <Story name="Basic">
    {names.map((name) => (
      <Avatar key={name} name={name} size={40} className="mr-2" />
    ))}
  </Story>
</Preview>

# Solid

Use the <code>isSolid</code> property to create a solid avatar.

<Preview>
  <Story name="Solid">
    {names.map((name) => (
      <Avatar key={name} isSolid name={name} size={40} className="mr-2" />
    ))}
  </Story>
</Preview>

# Sizes

Use the <code>size</code> property to change sizes. The initials will size automatically.

<Preview>
  <Story name="Sizes">
    <div>
      <Avatar
        isSolid
        marginRight={16}
        name={names[0]}
        size={16}
        className="mr-2"
      />
      <Avatar marginRight={16} name={names[1]} size={24} className="mr-2" />
      <Avatar marginRight={16} name={names[2]} size={32} className="mr-2" />
      <Avatar marginRight={16} name={names[3]} size={40} className="mr-2" />
      <Avatar marginRight={16} name={names[4]} size={96} className="mr-2" />
      <Avatar marginRight={16} name={names[5]} size={128} className="mr-2" />
    </div>
  </Story>
</Preview>

# Override color

Manually override the color using the <code>color</code> property.
Eg. <code>color="green"</code>.

<Preview>
  <Story name="Override color">
    {colors.map((color, index) => (
      <Avatar
        key={color}
        color={color}
        name={names[index]}
        size={40}
        className="mr-2"
      />
    ))}
  </Story>
</Preview>

# Solid with colors

You can use the <code>color</code> and <code>isSolid</code> properties in conjunction.

<Preview>
  <Story name="Solid with colors">
    {colors.map((color, index) => (
      <Avatar
        key={color}
        isSolid
        color={color}
        name={names[index]}
        size={40}
        className="mr-2"
      />
    ))}
  </Story>
</Preview>

# Dashed

The <code>isDashed</code> property creates a square avatar with a white background and dashed borders. Overrides <code>isSolid</code> and <code>color</code>.

<Preview>
  <Story name="Dashed">
    {names.map((name) => (
      <Avatar isDashed key={name} name={name} size={40} className="mr-2" />
    ))}
  </Story>
</Preview>

# Anonymous users

In cases where it's not possible to determine the name of the user, use the <code>hashValue</code> property to automatically determine the color.

<Preview>
  <Story name="Anonymous">
    {anonymousIds.map((id) => (
      <Avatar
        key={id}
        hashValue={id}
        name="Anonymous User"
        size={40}
        className="mr-2"
      />
    ))}
  </Story>
</Preview>
